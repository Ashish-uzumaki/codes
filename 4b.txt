MVI H,00H
MVI L,AAH 
// STACK IS OF 16 BIT AND PUSH H MEANS PUSH HL PAIR DATA IN STACK
PUSH H // ==(H+L) H=00H L=AAH
PUSH H 
MVI L,55H
PUSH H // ==(H+L) H=00H L==55H
PUSH H
MVI B,F5H
MVI C,8AH

// RESULT FOR PSW 55H
// ADDING 8A AND F5 AND FLAG REGISTER IS 55H
POP PSW // HERE USING POP WE CAN GET FLAG REGISTER AS 55H
// POP PSW MEANS (ACCUMULATOR+FLAG REGISTER)
MOV A,B // MOVE F5 (B REGISTER) TO A
ADC C // ADD C TO A
STA 6000H // STORING ADDING OPERATION IN 6000H
// SUBTRACTING  F5 AND 8A AND FLAG REGISTER IS 55H
POP PSW // HERE USING POP WE CAN GET FLAG REGISTER AS 55H
MOV A,B //
SBB C // SUBTRATING C FROM A
STA 6001H // 


// RESULT FOR PSW AAH
// ADDING 8A AND F5 AND FLAG REGISTER IS AAH
POP PSW // HERE USING POP WE CAN GET FLAG REGISTER AS AAH
MOV A,B  // MOVE B TO A
ADC C 
STA 6002H //
// // SUBTRACTING F5 AND 8A AND FLAG REGISTER IS 55H
POP PSW //
MOV A,B
SBB C
STA 6003H // SUBTRACT OPERATION IS STORED IN 6003H





// FOR PSW 55H //   ->  ADD RESULT IN 6000H   //- > SUB RESULT 6001H
// FOR PSW AAH //   ->  ADD RESULT IN 6002H   //- > SUB RESULT 6003H
// FOR 55H AUXILARY CAARY =1 AND CARRY FLAG=1 THEN THIS WILL CHANGE THE RESULT
// B=F5 AND C=8A 
// B= 1111 0101   AND C=1000 1010 and carryflag=1
// therefore in addition, A=B+C+1=80
// in subtraction, A=B-C-1=6A


// FOR AAH AUXILARY CAARY =0 AND CARRY FLAG=0 THEN THIS WILL CHANGE THE RESULT
// B=F5 AND C=8A 
// B= 1111 0101   AND C=1000 1010 and carryflag=0
// therefore in addition, A=B+C=7F
// in subtraction, A=B-C=6B
